## 14.2.2 TypeScript Configuration

A good `tsconfig.json` for strict TypeScript:

```json
{
  "compilerOptions": {
    "target": "ES2022",
    "module": "ESNext",
    "moduleResolution": "bundler",
    "lib": ["ES2022", "DOM", "DOM.Iterable"],
    
    "strict": true,
    "noUnusedLocals": true,
    "noUnusedParameters": true,
    "noImplicitReturns": true,
    "noFallthroughCasesInSwitch": true,
    "noUncheckedIndexedAccess": true,
    
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    
    "resolveJsonModule": true,
    "isolatedModules": true,
    
    "outDir": "./dist",
    "rootDir": "./src",
    
    "sourceMap": true,
    "declaration": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist", "tests"]
}
```

`"strict": true` enables all strict type checks. `"noUncheckedIndexedAccess": true` makes array access return `T | undefined`, preventing a common class of bugs.

## 14.2.3 Use TypeScript for New Projects

TypeScript catches bugs at compile time that JavaScript only catches at runtime. For any new project, use TypeScript.

```typescript
// TypeScript: errors caught at compile time
interface User {
  id: number;
  name: string;
  email: string;
}

function greet(user: User): string {
  return `Hello, ${user.name}`;
}

// greet({ name: "Alice" });  // Error: missing 'id' and 'email'
// greet("Alice");             // Error: string is not User

const alice: User = { id: 1, name: "Alice", email: "alice@example.com" };
console.log(greet(alice));  // OK
```

## 14.2.4 Modern JavaScript Features

Use modern syntax (ES2022+):

```javascript
// Optional chaining
const city = user?.address?.city;  // undefined if any step is undefined

// Nullish coalescing
const name = user.name ?? "Anonymous";  // only uses default for null/undefined

// Destructuring
const { name, email } = user;
const [first, ...rest] = items;

// Template literals
const greeting = `Hello, ${name}!`;

// Arrow functions
const square = (x) => x * x;

// Spread and rest
const merged = { ...obj1, ...obj2 };
const [first, ...others] = array;
function sum(...numbers) { return numbers.reduce((a, b) => a + b, 0); }

// for...of (for iterables)
for (const item of items) { ... }

// async/await
async function fetchData() {
  const response = await fetch('/api/data');
  return response.json();
}

// Map and Set
const userMap = new Map();
userMap.set(1, alice);
const uniqueTags = new Set(['js', 'ts', 'js']);  // {'js', 'ts'}
```

## 14.2.5 Common JavaScript Mistakes

### `==` vs `===`

```javascript
// BAD: == does type coercion
0 == false    // true
"" == 0       // true
null == undefined  // true

// GOOD: === checks type and value
0 === false   // false
"" === 0      // false
null === undefined  // false
```

### `var` vs `let` / `const`

```javascript
// BAD: var is function-scoped, hoisted
var x = 1;
if (true) {
    var x = 2;  // same x!
}
console.log(x);  // 2

// GOOD: let and const are block-scoped
let y = 1;
if (true) {
    let y = 2;  // different y
}
console.log(y);  // 1

// Use const by default; let only when reassignment is needed
const PI = 3.14;
let counter = 0;
```

### Async Error Handling

```javascript
// BAD: unhandled promise rejection
async function fetchUser(id) {
    const response = await fetch(`/api/users/${id}`);
    return response.json();
}
// If fetch fails, the rejection is unhandled

// GOOD: handle errors
async function fetchUser(id) {
    try {
        const response = await fetch(`/api/users/${id}`);
        if (!response.ok) {
            throw new Error(`HTTP ${response.status}`);
        }
        return response.json();
    } catch (error) {
        console.error(`Failed to fetch user ${id}:`, error);
        return null;
    }
}
```

### Mutating State

```javascript
// BAD: mutating objects
function addUser(users, user) {
    users.push(user);  // mutates the input
    return users;
}

// GOOD: return a new array
function addUser(users, user) {
    return [...users, user];  // immutable
}
```

## 14.2.6 Testing with Jest / Vitest

```typescript
// tests/userService.test.ts
import { describe, test, expect, beforeEach } from 'vitest';
import { UserService } from '../src/services/UserService';
import { MockUserRepository } from './mocks/MockUserRepository';

describe('UserService', () => {
  let service: UserService;
  let repo: MockUserRepository;

  beforeEach(() => {
    repo = new MockUserRepository();
    service = new UserService(repo);
  });

  describe('getUser', () => {
    test('returns user when found', async () => {
      repo.setUsers([{ id: 1, name: 'Alice' }]);
      
      const user = await service.getUser(1);
      
      expect(user).toEqual({ id: 1, name: 'Alice' });
    });

    test('throws when user not found', async () => {
      repo.setUsers([]);
      
      await expect(service.getUser(1)).rejects.toThrow('User not found');
    });
  });
});
```

## 14.2.8 Key Takeaways

- Use TypeScript for all new projects; enable `strict` mode.
- Use modern JS features: optional chaining, nullish coalescing, destructuring, async/await.
- Use `===` (not `==`), `let`/`const` (not `var`), `const` by default.
- Handle async errors properly with try/catch.
- Avoid mutating inputs; return new objects.
- Tools: TypeScript (types), ESLint (lint), Prettier (format), Jest/Vitest (tests), Vite (build).
